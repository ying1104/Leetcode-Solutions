#### 1525. Number of Good Ways to Split a String

```java
// My first thought
class Solution {
    public int numSplits(String s) {
        int count = 0;
        int n = s.length();
        
        int[] left = new int[26];
        int[] right = new int[26];
        
        for (int i = 0; i < n; i++) {
            right[s.charAt(i) - 'a']++;
        }
        
        for (int i = 0; i < n; i++) {
            char c = s.charAt(i);
            left[c - 'a']++;
            right[c - 'a']--;
            if (isEqual(left, right)) {
                count++;
            }
        }
        return count;
    }
    
    private boolean isEqual(int[] left, int[] right) {
        int count1 = 0;
        int count2 = 0;
        for (int i = 0; i < 26; i++) {
            if (left[i] != 0) count1++;
            if (right[i] != 0) count2++;
        }
        return count1 == count2;
    }
}
```
