#### 206. Reverse Linked List

```java
class Solution {
    // My solution
    public ListNode reverseList(ListNode head) {
        if (head == null || head.next == null) {
            return head;
        }
        
        ListNode prev = head.next;
        ListNode temp = head.next.next;
        head.next = null;
        prev.next = head;
        head = prev;
        while (temp != null) {
            prev = temp;
            temp = temp.next;
            prev.next = head;
            head = prev;
        }
        return head;
    }
}
```
