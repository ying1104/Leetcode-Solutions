#### 261. Graph Valid Tree

```java
// BFS using queue
class Solution {
    public boolean validTree(int n, int[][] edges) {
        // No cycle or isolated leaves in the tree makes it valid?
        List<Set<Integer>> list = new ArrayList<>();
        for (int i = 0; i < n; i++)
            list.add(new HashSet<>());
        
        // store the edges into list
        for (int i = 0; i < edges.length; i++) {
            int p1 = edges[i][0];
            int p2 = edges[i][1];
            list.get(p1).add(p2);
            list.get(p2).add(p1);
        }
        
        boolean[] visited = new boolean[n];
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(0);
        
        while (!queue.isEmpty()) {
            int cur = queue.poll();
            if (visited[cur]) return false;
            visited[cur] = true;
            Set<Integer> nodes = list.get(cur);
            for (int node : nodes) { 
                
                list.get(node).remove(cur);
                queue.add(node);
            }
            
        }
        for (boolean v : visited)
            if (!v) return false;
        return true;
    }
}
```

```JAVA
// DFS using stack
class Solution {
    public boolean validTree(int n, int[][] edges) {
        // No cycle or isolated leaves in the tree makes it valid?
        List<Set<Integer>> list = new ArrayList<>();
        for (int i = 0; i < n; i++)
            list.add(new HashSet<>());
        
        // store the edges into list
        for (int i = 0; i < edges.length; i++) {
            int p1 = edges[i][0];
            int p2 = edges[i][1];
            list.get(p1).add(p2);
            list.get(p2).add(p1);
        }
        
        boolean[] visited = new boolean[n];
        Stack<Integer> stack = new Stack<>();
        stack.push(0);
        while (!stack.isEmpty()) {
            int cur = stack.pop();
            if (visited[cur]) return false;
            visited[cur] = true;
            for (int node : list.get(cur)) {
                if (!visited[node])
                    stack.push(node);
            }
        }
        
        for (boolean b : visited) {
            if (!b) return false;
        }
        return true;

    }
    

}
```
