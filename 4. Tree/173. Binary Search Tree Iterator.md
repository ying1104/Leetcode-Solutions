#### 173. Binary Search Tree Iterator

```java
class BSTIterator {
    Stack<TreeNode> stack = new Stack<>();
    TreeNode root;
    public BSTIterator(TreeNode root) {
        this.root = root;
    }
    
    public int next() {
        while (root != null) {
            stack.push(root);
            root = root.left;
        }
        root = stack.pop();
        int res = root.val;
        root = root.right;
        return res;
    }
    
    public boolean hasNext() {
        // System.out.println(stack.size());
        return (!stack.isEmpty() || root != null);
    }
}
```
