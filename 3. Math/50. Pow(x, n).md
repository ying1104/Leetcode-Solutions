#### 50. Pow(x, n)

```java
// Solution 1: Time exceeds for large numbers
class Solution {
    public double myPow(double x, int n) {
        if (n == 0) return 1;
        // if (x == 0 && n < 0) 
        //     throw new Exception("Error inputs");
        if (n < 0) {
            n = -n;
            x = 1 / x;
        }
        double res = 1;
        while (n > 0) {
            n--;
            res *= x;
        }
        return res;
    }
}
```

```java
class Solution {
    public double myPow(double x, int n) {
        if (n == 0) return 1;
        // if (x == 0 && n < 0) 
        //     throw new Exception("Error inputs");
        if (n == Integer.MIN_VALUE) {
            n = n / 2;
            x = x * x;
        }
        if (n < 0) {
            n = -n;
            x = 1 / x;
        }

        if (n % 2 == 0) return myPow(x * x, n / 2);
        else return x * myPow(x * x, n / 2);
        
    }
}
```
